# ============================================
# Frontend Dependencies
# ============================================

# MapTiler SDK for maps
# https://docs.maptiler.com/sdk-js/

# Chart.js for statistics visualization
# https://www.chartjs.org/

# Axios for API requests (optional, can use fetch)
# https://axios-http.com/

# ============================================
# Installation via CDN (recommended for MVP)
# ============================================

# Add these to your HTML:
# <script src="https://cdn.maptiler.com/maptiler-sdk-js/latest/maptiler-sdk.umd.js"></script>
# <link href="https://cdn.maptiler.com/maptiler-sdk-js/latest/maptiler-sdk.css" rel="stylesheet" />
# <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

# ============================================
# Or install via npm (if using build tools)
# ============================================

# npm install @maptiler/sdk
# npm install chart.js
# npm install axios
